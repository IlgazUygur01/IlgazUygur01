Project 1: Weather Advisory

This Swift project gives a weather-based recommendation on what item to take with you based on the forecast.


// Swift code for Weather Advisory
let weather = "rain"

if weather == "rain" {
    print("Take an umbrella with you!")
} else if weather == "snow" {
    print("Take a scarf with you!")
} else if weather == "sunny" {
    print("Don't forget your sunglasses!")
} else {
    print("No special item recommendation.")
}

Project 2: Grade Calculator

This project calculates the average of two grades and determines whether the student has passed or failed.


// Swift code for Grade Calculator
let grade1 = 70
let grade2 = 40

let average = (grade1 + grade2) / 2

if average >= 50 {
    print("Passed")
} else {
    print("Failed")
}
