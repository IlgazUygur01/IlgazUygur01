Project 3: Task Management App

This is a to-do list app built with SwiftUI, where users can manage their tasks efficiently.


// Swift code for Task Management App
import SwiftUI

struct ContentView: View {
    @State private var tasks = [String]()
    @State private var newTask = ""

    var body: some View {
        VStack {
            TextField("Enter a new task", text: $newTask)
            Button("Add Task") {
                tasks.append(newTask)
                newTask = ""
            }
            List(tasks, id: \.self) { task in
                Text(task)
            }
        }
    }
}
