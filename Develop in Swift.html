Project 5: Rectangle Area

This Swift project calculates the area of a rectangle using length and width.


func rectangleArea(length: Int, width: Int) -> Int {
    let area = length * width
    return area
}

let area = rectangleArea(length: 10, width: 5)
print("The area of the rectangle is: \(area)")

Project 6: Current Date and Time

This project returns the current date and time as a string.


import Foundation

func currentDateAndTime() -> String {
    let date = Date()
    return "\(date)"
}

let currentDate = currentDateAndTime()
print("Current date and time: \(currentDate)")

Project 7: Random Number Generator

This Swift project generates a random number between 1 and 100.


func randomNumber() -> Int {
    return Int.random(in: 1...100)
}

let number = randomNumber()
print("Random number: \(number)")
