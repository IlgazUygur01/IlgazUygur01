Project 1: Weather Advisory

This Swift project gives a weather-based recommendation on what item to take with you based on the forecast.


// Swift code for Weather Advisory
let weather = "rain"

if weather == "rain" {
    print("Take an umbrella with you!")
} else if weather == "snow" {
    print("Take a scarf with you!")
} else if weather == "sunny" {
    print("Don't forget your sunglasses!")
} else {
    print("No special item recommendation.")
}

Project 2: Grade Calculator

This project calculates the average of two grades and determines whether the student has passed or failed.


// Swift code for Grade Calculator
let grade1 = 70
let grade2 = 40

let average = (grade1 + grade2) / 2

if average >= 50 {
    print("Passed")
} else {
    print("Failed")
}

Project 4: Simple Calculator

This Swift project performs basic arithmetic operations such as addition, subtraction, division, and multiplication, and displays the result.


// Swift code for Simple Calculator
var number1 = 5
var number2 = 10

let addition = number1 + number2
print("The sum of these two numbers: \(addition)")

let subtraction = number1 - number2
print("The difference of these two numbers: \(subtraction)")

let division = number1 / number2
if number2 == 0 {
    print("Number 2 cannot be zero")
} else {
    print("The result of dividing the first number by the second:  \(division)")
}

let multiplication = number1 * number2
print ("The product of these two numbers: \(multiplication)")

let remainder = number1 % number2
print("The remainder when the first number is divided by the second: \(remainder)")

for i in 1...addition {
    print(i)
}
